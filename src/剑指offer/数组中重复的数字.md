#### 剑指offer第三题 数组中重复的数字



~~~javascript
const findRepeatNumber = function(nums){
    let len = nums.length
    for(let i = 0;i<len;i++){
        while(nums[i] != i){
            if(nums[i] === nums[nums[i]]){
                return nums[i]
            }
            let temp = nums[i]
        	nums[i] = nums[temp]
        	nums[temp] = temp  
        }  
    }
    return -1
}
~~~
