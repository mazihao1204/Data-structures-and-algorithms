### 初级排序 n*n
1 选择排序  Selection Sort
  每次找最小值，然后放在待排序数组的起始位置
~~~ js
function selectionSort(arr){
  let len = arr.length
  let minIndex,temp
  for(let i = 0;i<len-1;i++){
    minIndex = i
    for(let j = i + 1;j < len;j++){
      if(arr[j] < arr[minIndex]){
        minIndex = j
      }
    }
    temp = arr[i]
    arr[i] = arr[minIndex]
    arr[minIndex] = temp
  }
  return arr
}
~~~
